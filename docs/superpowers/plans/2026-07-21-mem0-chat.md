# mem0-chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Create a new `functions/mem0-chat/` edge function with persistent memory via mem0 and an agentic flow where MiniMax orchestrates via tool calls.

**Architecture:** MiniMax recibe un system prompt con tools definidas (save_user_profile, generate_routine, save_routine). Responde con texto o con JSON tool call. El handler orquesta el loop: mem0 → MiniMax → tool execution → MiniMax → respuesta final. Mem0 guarda perfil y facts entre sesiones.

**Tech Stack:** Deno, MiniMax API (direct fetch), mem0 standalone (`npm:@mem0/vercel-ai-provider`), InsForge SDK (`npm:@insforge/sdk`), Zod.

## Global Constraints

- Sin tests unitarios
- Sin templates hardcodeados de respuesta
- El flujo lo gobierna MiniMax vía tool calls, no código
- Todas las herramientas (CORS, auth, rate limit) se copian de `functions/minimax-chat/`
- `routines.ts` importa directamente de `functions/minimax-chat/routine.ts` y `functions/minimax-chat/persistence.ts`
- Deno, edge function de InsForge
- `MEM0_API_KEY` como env var
- Tabla `chat_messages` con pgvector (embeddings embo-01, 1024 dimensiones)
- Embeddings solo para mensajes de usuario (no assistant/tool)

```ts
import { RATE_LIMIT, RATE_WINDOW, rateLimitStore } from './constants.ts';

export function extractUserIdFromToken(authHeader: string | null): string | null {
  if (!authHeader || !authHeader.startsWith('Bearer ')) return null;
  try {
    const token = authHeader.replace('Bearer ', '');
    const parts = token.split('.');
    if (parts.length !== 3) return null;
    const payload = JSON.parse(atob(parts[1]));
    return payload.sub || payload.user_id || null;
  } catch {
    return null;
  }
}

export function checkRateLimit(userId: string): boolean {
  const now = Date.now();
  const record = rateLimitStore.get(userId);
  if (!record || now > record.resetAt) {
    rateLimitStore.set(userId, { count: 1, resetAt: now + RATE_WINDOW });
    return true;
  }
  if (record.count >= RATE_LIMIT) return false;
  record.count++;
  return true;
}

export function buildCorsHeaders(origin: string): Record<string, string> {
  return {
    'Access-Control-Allow-Origin': origin,
    'Access-Control-Allow-Methods': 'POST, OPTIONS',
    'Access-Control-Allow-Headers': 'Content-Type, Authorization, apikey',
    'Access-Control-Allow-Credentials': 'true',
  };
}

export function parseMiniMaxJson(text: string): unknown {
  return JSON.parse(
    text.replace(/<think>[\s\S]*?<\/think>/gi, '')
      .replace(/```json|```/gi, '')
      .trim(),
  );
}

export function stripThinkTags(delta: string, initialInThink: boolean): { cleaned: string; inThink: boolean } {
  let remaining = true;
  let inThink = initialInThink;
  let result = '';
  let cursor = delta;
  while (remaining && cursor.length > 0) {
    if (inThink) {
      const end = cursor.indexOf('</think>');
      if (end === -1) {
        remaining = false;
      } else {
        cursor = cursor.slice(end + 8);
        inThink = false;
      }
    } else {
      const start = cursor.indexOf('<think>');
      if (start === -1) {
        result += cursor;
        remaining = false;
      } else {
        result += cursor.slice(0, start);
        cursor = cursor.slice(start + 7);
        inThink = true;
      }
    }
  }
  return { cleaned: result, inThink };
}
```

```ts
const mmKey = Deno.env.get('MINIMAX_API_KEY') ?? '';

export async function generateEmbedding(text: string): Promise<number[] | null> {
  try {
    const res = await fetch('https://api.minimax.io/v1/embeddings', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${mmKey}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ model: 'embo-01', texts: [text], type: 'db' }),
    });
    if (!res.ok) return null;
    const data = await res.json();
    if (data?.base_resp?.status_code !== 0) return null;
    return data.vectors?.[0] ?? null;
  } catch {
    return null;
  }
}
```

- [ ] **Step 2: Create the SQL function for pgvector search (run in InsForge)**

```sql
CREATE OR REPLACE FUNCTION search_chat_messages(
  p_user_id uuid,
  p_embedding vector(1024),
  p_limit int DEFAULT 5
)
RETURNS TABLE (content text, role text, created_at timestamptz, similarity float)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT cm.content, cm.role, cm.created_at,
         1 - (cm.embedding <=> p_embedding) AS similarity
  FROM chat_messages cm
  WHERE cm.user_id = p_user_id
    AND cm.embedding IS NOT NULL
  ORDER BY cm.embedding <=> p_embedding
  LIMIT p_limit;
END;
$$;
```

```ts
import type { Profile } from './schemas.ts';

const MEM0_API_KEY = Deno.env.get('MEM0_API_KEY') ?? '';

interface MemoryFact {
  text: string;
  type: 'profile' | 'routine' | 'preference';
}

function serializeProfile(profile: Profile): string {
  return `User profile — age:${profile.age}, weight:${profile.weightKg}kg, goal:${profile.goal}, level:${profile.level}, daysPerWeek:${profile.daysPerWeek}`;
}

export async function loadUserProfile(userId: string): Promise<Profile | null> {
  try {
    const results = await searchMemories(`user profile ${userId}`, {
      user_id: userId,
      mem0ApiKey: MEM0_API_KEY,
    });
    const profileFact = (results as MemoryFact[]).find((r) => r.type === 'profile');
    if (!profileFact?.text) return null;
    return parseProfileFromText(profileFact.text);
  } catch (e) {
    console.error('mem0 search error', e);
    return null;
  }
}

export async function saveUserProfile(userId: string, profile: Profile): Promise<void> {
  try {
    await addMemories(
      [{ role: 'user', content: [{ type: 'text', text: serializeProfile(profile) }] }],
      { user_id: userId, mem0ApiKey: MEM0_API_KEY },
    );
  } catch (e) {
    console.error('mem0 add error', e);
  }
}

export async function saveRoutineMemory(userId: string, routineName: string): Promise<void> {
  try {
    await addMemories(
      [{ role: 'user', content: [{ type: 'text', text: `Last routine created: ${routineName}` }] }],
      { user_id: userId, mem0ApiKey: MEM0_API_KEY },
    );
  } catch (e) {
    console.error('mem0 add error', e);
  }
}

function parseProfileFromText(text: string): Profile | null {
  const age = extractInt(text, /age:(\d+)/);
  const weightKg = extractFloat(text, /weight:([\d.]+)/);
  const goal = extractEnum(text, /goal:(\w+)/, ['strength', 'cardio', 'fat_loss', 'general']);
  const level = extractEnum(text, /level:(\w+)/, ['beginner', 'intermediate', 'advanced']);
  const daysPerWeek = extractInt(text, /daysPerWeek:(\d+)/);
  if (age === null || weightKg === null || !goal || !level || daysPerWeek === null) return null;
  return { age, weightKg, goal, level, daysPerWeek };
}
```

```ts
import { routineProfileSchema } from './schemas.ts';
import { loadUserProfile, saveUserProfile, saveRoutineMemory } from './mem0.ts';
import { generateRoutineProposal } from '../minimax-chat/routine.ts';
import { saveRoutineToDb } from '../minimax-chat/persistence.ts';
import { createClient } from 'npm:@insforge/sdk';
import type { Profile, ApprovedRoutine } from './schemas.ts';

const URL_BASE = Deno.env.get('INSFORGE_URL') ?? 'https://4af6r2tm.eu-central.insforge.app';

export async function executeTool(
  toolName: string,
  args: Record<string, unknown>,
  context: { userId: string; token: string; anonKey: string },
): Promise<{ result: unknown; stateChange: boolean }> {
  switch (toolName) {
    case 'save_user_profile':
      return handleSaveProfile(args, context.userId);
    case 'generate_routine':
      return handleGenerateRoutine(context);
    case 'save_routine':
      return handleSaveRoutine(args, context);
    default:
      return { result: { error: `Unknown tool: ${toolName}` }, stateChange: false };
  }
}

async function handleSaveProfile(
  args: Record<string, unknown>,
  userId: string,
): Promise<{ result: unknown; stateChange: boolean }> {
  const parsed = routineProfileSchema.safeParse(args);
  if (!parsed.success) {
    return {
      result: { error: 'Datos de perfil inválidos. Asegúrate de tener: age (14-99), weightKg (40-120), goal, level, daysPerWeek (1-7).' },
      stateChange: false,
    };
  }
  await saveUserProfile(userId, parsed.data);
  return {
    result: { success: true, profile: parsed.data },
    stateChange: true,
  };
}

async function handleGenerateRoutine(
  context: { userId: string; token: string; anonKey: string },
): Promise<{ result: unknown; stateChange: boolean }> {
  const profile = await loadUserProfile(context.userId);
  if (!profile) {
    return {
      result: { error: 'No tengo el perfil completo. Pregúntale al usuario su edad, peso, objetivo, nivel y días de entrenamiento.' },
      stateChange: false,
    };
  }
  const proposal = await generateRoutineProposal(profile, context.token, context.anonKey);
  return { result: { proposal }, stateChange: false };
}

async function handleSaveRoutine(
  args: Record<string, unknown>,
  context: { userId: string; token: string; anonKey: string },
): Promise<{ result: unknown; stateChange: boolean }> {
  const client = createClient({ baseUrl: URL_BASE, anonKey: context.anonKey });
  client.setAccessToken(context.token);
  const id = await saveRoutineToDb(client, context.userId, args as unknown as ApprovedRoutine);
  await saveRoutineMemory(context.userId, (args as { name?: string }).name ?? '');
  return { result: { id, success: true }, stateChange: true };
}
```

```ts
import { loadUserProfile } from './mem0.ts';
import { executeTool } from './tools.ts';
import { generateEmbedding } from './embeddings.ts';
import { SYSTEM_PROMPT } from './constants.ts';
import { createClient } from 'npm:@insforge/sdk';

const mmKey = Deno.env.get('MINIMAX_API_KEY') ?? '';
const URL_BASE = Deno.env.get('INSFORGE_URL') ?? 'https://4af6r2tm.eu-central.insforge.app';
const MAX_TOOL_LOOP = 5;
const HISTORY_LIMIT = 50;

export async function handleChatRequest(
  userId: string,
  token: string,
  anonKey: string,
  message: string,
  corsHeaders: Record<string, string>,
): Promise<Response> {
  const client = createClient({ baseUrl: URL_BASE, anonKey });
  client.setAccessToken(token);

  try {
    const profile = await loadUserProfile(userId);
    const history = await loadChatHistory(client, userId, message);

    const profileContext = profile
      ? `\n\nPerfil del usuario: ${JSON.stringify(profile)}\n`
      : '\n\nEl usuario no tiene perfil guardado. Pregúntale sus datos si los necesitas.\n';

    const historyContext = history.length > 0
      ? `\nHistorial reciente:\n${history.map((m) => `${m.role}: ${m.content}`).join('\n')}\n`
      : '';

    const messages: Array<{ role: string; content: string }> = [
      { role: 'system', content: SYSTEM_PROMPT + profileContext + historyContext },
      { role: 'user', content: message },
    ];

    await saveMessage(client, userId, 'user', message);

    let toolCalls = 0;
    let finalResponse = '';

    while (toolCalls < MAX_TOOL_LOOP) {
      const mmRes = await fetch('https://api.minimax.io/v1/chat/completions', {
        method: 'POST',
        headers: { 'Authorization': `Bearer ${mmKey}`, 'Content-Type': 'application/json' },
        body: JSON.stringify({ model: 'MiniMax-M2.7', messages, temperature: 0.7 }),
      });

      if (!mmRes.ok) {
        const errText = await mmRes.text();
        return new Response(
          JSON.stringify({ error: `MiniMax API error: ${mmRes.status} ${errText}` }),
          { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
        );
      }

      const mmData = await mmRes.json();
      const text = mmData?.choices?.[0]?.message?.content ?? '';
      const parsed = tryParseToolCall(text);

      if (!parsed || !parsed.tool) {
        finalResponse = text;
        await saveMessage(client, userId, 'assistant', text);
        return new Response(JSON.stringify({ response: text }), {
          status: 200,
          headers: { ...corsHeaders, 'Content-Type': 'application/json' },
        });
      }

      const { result, stateChange } = await executeTool(
        parsed.tool,
        parsed.args ?? {},
        { userId, token, anonKey },
      );

      messages.push(
        { role: 'assistant', content: text },
        { role: 'user', content: `Resultado de ${parsed.tool}: ${JSON.stringify(result)}` },
      );

      toolCalls++;

      if (stateChange && parsed.tool === 'save_routine') {
        finalResponse = `Rutina guardada correctamente.`;
        await saveMessage(client, userId, 'assistant', finalResponse);
        return new Response(JSON.stringify({ response: finalResponse }), {
          status: 200,
          headers: { ...corsHeaders, 'Content-Type': 'application/json' },
        });
      }
    }

    return new Response(
      JSON.stringify({ error: 'Demasiadas operaciones. Intenta de nuevo.' }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  } catch (e) {
    const errorMessage = e instanceof Error ? e.message : String(e);
    console.error('handler error', e);
    return new Response(
      JSON.stringify({ error: errorMessage }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }
}

async function loadChatHistory(
  client: ReturnType<typeof createClient>,
  userId: string,
  currentMessage: string,
): Promise<Array<{ role: string; content: string }>> {
  const { data } = await client.database
    .from('chat_messages')
    .select('role, content')
    .eq('user_id', userId)
    .order('created_at', { ascending: false })
    .limit(HISTORY_LIMIT);

  const recent = (data ?? []).reverse();
  return recent as Array<{ role: string; content: string }>;
}

async function saveMessage(
  client: ReturnType<typeof createClient>,
  userId: string,
  role: string,
  content: string,
): Promise<void> {
  try {
    const embedding = role === 'user' ? await generateEmbedding(content) : null;
    await client.database.from('chat_messages').insert([{
      user_id: userId,
      role,
      content,
      embedding: embedding ?? undefined,
    }]);
  } catch (e) {
    console.error('save error', e);
  }
}

function tryParseToolCall(text: string): { tool: string; args?: Record<string, unknown> } | null {
  try {
    const cleaned = text.replace(/<think>[\s\S]*?<\/think>/gi, '').replace(/```json|```/gi, '').trim();
    const parsed = JSON.parse(cleaned);
    if (parsed && typeof parsed.tool === 'string') {
      return { tool: parsed.tool, args: typeof parsed.args === 'object' ? parsed.args : {} };
    }
    return null;
  } catch {
    return null;
  }
}
```

```ts
import { buildCorsHeaders, extractUserIdFromToken, checkRateLimit } from './utils.ts';
import { handleChatRequest } from './handler.ts';

export default async function (req: Request): Promise<Response> {
  const origin = req.headers.get('Origin')!;
  const corsHeaders = buildCorsHeaders(origin);

  if (req.method === 'OPTIONS') {
    return new Response(null, { status: 204, headers: corsHeaders });
  }

  const authHeader = req.headers.get('Authorization');
  const userId = extractUserIdFromToken(authHeader);

  if (!userId) {
    return new Response(
      JSON.stringify({ error: 'Authentication required' }),
      { status: 401, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  if (!checkRateLimit(userId)) {
    return new Response(
      JSON.stringify({ error: 'Rate limit exceeded. Try again later.' }),
      { status: 429, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  const { message } = await req.json();

  if (!message || typeof message !== 'string') {
    return new Response(
      JSON.stringify({ error: 'Message is required' }),
      { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  const token = authHeader!.slice(7);
  const anonKey = req.headers.get('apikey') ?? '';

  return handleChatRequest(userId, token, anonKey, message, corsHeaders);
}
```
